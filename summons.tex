		(\signX * 75, 0) coordinate(CsignX)
		(CsignX |- CsignY) coordinate(x)

		(x)node[
			anchor=south,
			text width=60mm,
			align=center
		]{\signName{#1}}

		(x)++(0,-1)node[
			anchor=base,
			text width=60mm,
			align=center
		]{\signDesc{#3}}

		(x)++(0,4.5)node{\begin{tikzpicture}[
			scale=0.75,
			ultra thick,
			line cap=round,
			line join=round
		]
			\InputIfFileExists{\texname}{}{}
		\end{tikzpicture}}
		;
	
	\addtolength{\signX}{1mm}
}
