\newcommand{\xsign}[2][0mm]{
	\stepcounter{SignColumn}
	\caselower[q]{#2}\edef\texname{signs/\thestring.tex}
	\edef\columnName{Column\theSignColumn}
	%
	\draw
		(\columnName) coordinate(Column)
		(Column |- Baseline) coordinate(SignHere)
		(SignHere)++(0,#1)node{\begin{tikzpicture}[
			scale=0.75,
			ultra thick,
			line cap=round,
			line join=round
		]
			\InputIfFileExists{\texname}{}{}
		\end{tikzpicture}}
		;
}

	\xsign[40mm]{#2}
	%

		(SignHere)node[
			anchor=south,
			text width=55mm,
			align=center
		]{\signName{#1}}

		(SignHere)++(0,-0mm)node[
			anchor=north,
			text width=55mm,
			align=center
		]{\signDesc{#3}}
		;
}
