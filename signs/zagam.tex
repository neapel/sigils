	\foreach \XS in {-1,1} {[xscale=\XS]
		let \n0={1.225} in
			(0,0) -- ++(\n0,0) coordinate(v)
			-- ++(\n0,0)
			let \n1={0.5} in ++(\n1,0) circle(\n1)

		(v) -- ++(0,0.85) coordinate(h) -- ++(0,1.5)
		let \n1={0.2} in ++(0,\n1) circle(\n1)

		\foreach \XZ in {-1,1} {[xscale=\XZ]
			(h) -- ++(0.9,0)
			let \n1={0.45} in ++(0,-\n1) -- ++(0,2 * \n1)
		}
	}

	{[rotate=-240]
		let \n0={30}, \n1={1} in
			(0,0) arc[start angle=\n0, end angle={180-\n0}, radius=\n1]
			arc[start angle=-\n0, delta angle={-180 - \n0}, radius={0.5 * \n1}]

			(0,0) ++(180+\n0:\n1) ++(90:\n1) -- ++(0,1.5) coordinate(t)
			let \n3={0.4} in
				(t) ++(-\n3,-\n3) -- (t) -- ++(\n3,-\n3) -- cycle
	}
	;
