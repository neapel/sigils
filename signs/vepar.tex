\draw
	let \n0={2.5}, \n1={0.45}, \n2={0.3} in
		(\n0,0) arc[start angle=0, end angle=-180, radius=\n0]
		coordinate(x) {[rotate=158]
			(x) -- ++(0,2.65)
			arc[start angle=0, end angle=160, radius=0.5]
		}

		(\n0 - \n1 - \n2,0) circle(\n2)
		(-\n0 - \n1 - \n2 - 0.2,0) circle(\n2)
		++(\n2,0) -- (\n0 + \n1,0)
		++(\n2,0) circle(\n2)

	let \n0={0.35}, \n1={1.8}, \n2={0.5} in
		(-0.5,0) coordinate(t)
		(t) -- ++(0,\n1 + \n0) coordinate(r)
		(r) ++(-\n0,\n0) -- (r) -- ++(\n0,\n0) -- cycle

		\foreach \XS in {-1,1} {[xscale=\XS]
			(t) ++(1,0) -- ++(0,\n1) coordinate(x) -- ++(0,\n0)
			(x) ++(-\n0,0) -- ++(2 * \n0,0)
		}

		(1,0) -- ++(0,-\n1)
		(-1,0) -- ++(0,-\n1)
		(1 + \n2,-0.5 * \n1) -- (-1 - \n2,-0.5 * \n1)
		(0,-0.5 * \n1) -- (0,-\n1) coordinate(x) -- ++(0,-\n0)
		(x) ++(-\n0,0) -- ++(2 * \n0,0)

	;
